interface MissingElement {

  presentElements: PresentElement[];

  missingElements,

        <h4 className="text-sm font-semibold text-gray-700 mb-3">Missing:</h4>

          {missingElements.map((element, index) => (
            <div
              key={index}
              className="flex justify-between items-center py-3 px-4 bg-gray-50 rounded-md hover:bg-gray-100 transition-colors"
            >
              <div className="flex items-center gap-3">
                <span className="text-gray-400 text-lg">âœ—</span>

    missing: Array<{
      name: string;

      priority: number;
    }>;
    present: Array<{
      name: string;
    }>;
  };

Data Processing
// When fetching analysis results
const processAnalysisResults = (data: AnalysisResults) => {
  // Extract missing elements
  const missingElements = data.missingElements.missing;
  const presentElements = data.missingElements.present;

  // Sort missing elements by impact (high â†’ medium â†’ low)
  const sortedMissing = missingElements.sort((a, b) => {
    const impactOrder = { high: 0, medium: 1, low: 2 };
    return impactOrder[a.impact] - impactOrder[b.impact];
  });

  return {
    missingElements: sortedMissing,
    presentElements,
    recommendations: data.recommendations,
  };
};
